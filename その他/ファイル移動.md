```bat
@echo off
chcp 65001 > nul
setlocal enabledelayedexpansion
@rem \で区切ること。最後の\は不要。
set "from="
set "to_1="
set "to_2="
set "log_file_dir="

xcopy /s /e /y /i "!from!" "!to_1!"
xcopy /s /e /y /i "!from!" "!to_2!"

@REM rd /s /q "!from!"
@REM 現在の日時からファイル名を作成してログファイルを作成する。ファイル名はYYYYMMDD_HHMMSS.log　0埋めする。例：20200101_013512.log

set yyyy=%date:~0,4%
set mm=%date:~5,2%
set dd=%date:~8,2%
set TMPTIME=%time: =0%
set hh=%TMPTIME:~0,2%
set min=%TMPTIME:~3,2%
set ss=%TMPTIME:~6,2%
set logfile=%log_file_dir%\%yyyy%%mm%%dd%_%hh%%min%%ss%.log

@REM to_2のファイル一覧をログファイルに出力する。
dir /b /s "!to_2!" > "!logfile!"
```
